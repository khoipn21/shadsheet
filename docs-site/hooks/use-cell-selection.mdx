# useCellSelection

Handles cell selection and range extension.

## Usage

```tsx
import { useCellSelection } from "@shadsheet/ui";

const { handleCellMouseDown, handleCellMouseEnter, handleMouseUp } = useCellSelection({
  activeCell,
  setActiveCell,
  setSelection,
  visibleColumnIds,
  mergedCells,
  mergedCellLookup,
});
```
